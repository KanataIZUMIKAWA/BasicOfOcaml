int add(int x, int y){
    int result = x + y;
    return result;
}

1) Normal
[Register]
int add(int %eax, int %ebx){
    int %ecx = %eax + %ebx;
    return %ecx;
}

[Assembly]
add:
    mov %edi, %eax;
    mov %esi, %ebx;
    mov %eax, %ecx;
    add %ebx, %ecx;
    mov %ecx, %eax;
    ret

int add(int %edi, int %esi){

    return %eax;
}

add:
    mov %edi, %eax
    add %esi, %eax
